해결하신 분의 풀이

```javascript
const fs = require("fs");
const input = fs.readFileSync("/dev/stdin").toString().split("");
const n = parseInt(input[0]);
const k = parseInt(input[1]);
const arr = [];
const answer = "";
const deleted = [];
for (let i = 0; i < n; i++) {
  arr[i] = i + 1;
}
for (j = 0; j < n; j++) {
  for (let m = 1; m < k; m++) {
    if (m === k) {
      deleted.push(arr.shift());
    } else {
      arr.push(arr.shift());
    }
  }
}
answer = "<" + deleted.join(",") + ">";

```
